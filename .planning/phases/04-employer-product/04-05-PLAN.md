```
import type { APIContext } from "astro";
import { getEnv } from "../../lib/env";

function isSafeRedirect(url: string): boolean {
  // Allow absolute http(s) URLs (for ATS apply links) and same-origin paths
  if (url.startsWith("/")) return true;
  try {
    const parsed = new URL(url);
    return parsed.protocol === "http:" || parsed.protocol === "https:";
  } catch {
    return false;
  }
}

export async function GET({ request, locals, redirect }: APIContext) {
  const niche = locals.niche;
  const env = getEnv(locals);
  const url = new URL(request.url);

  const jobId = url.searchParams.get("job") ?? "";
  const type = url.searchParams.get("type") ?? "view";
  const employerId = url.searchParams.get("employer") ?? "";
  const redirectTarget = url.searchParams.get("redirect");

  if (!jobId) {
    return Response.json({ error: "Missing job id" }, { status: 400 });
  }

  // Pitfall 4 — local dev does not simulate Analytics Engine
  const analytics = (env as { ANALYTICS?: AnalyticsEngineDataset }).ANALYTICS;
  if (analytics && typeof analytics.writeDataPoint === "function") {
    try {
      analytics.writeDataPoint({
        blobs: [jobId, type, niche.id, employerId],
        doubles: [1],
        indexes: [jobId],
      });
    } catch {
      // writeDataPoint is documented as fire-and-forget; swallow defensively
    }
  }

  if (type === "apply") {
    if (!redirectTarget) {
      return Response.json({ error: "Missing redirect target" }, { status: 400 });
    }
    if (!isSafeRedirect(redirectTarget)) {
      return Response.json({ error: "Invalid redirect target" }, { status: 400 });
    }
    return redirect(redirectTarget, 302);
  }

  return new Response(null, { status: 204 });
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public visitor → /api/track | Unauthenticated; attacker can spoof events but only inflates their own employer's counts (low value) |
| Public visitor → /api/track redirect | Open-redirect surface; safe-list http(s) and same-origin only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-22 | Tampering | /api/track redirect param | mitigate | `isSafeRedirect()` rejects non-http(s) and non-`/` URLs (blocks `javascript:`, `data:`, etc.) |
| T-04-23 | Spoofing | Event spoofing — attacker fires fake view/apply events | accept | Counts are advisory; no billing decisions or rate-limits gated on them; future Phase 5 may add Turnstile |
| T-04-24 | Information Disclosure | Job ID enumeration via /api/track | accept | Job IDs are already public on /jobs and in sitemap; not a secret |
| T-04-25 | Denial of Service | Spam writes to Analytics Engine | accept | CF Analytics Engine has generous free tier; no DB pressure (no Supabase write per event) |
</threat_model>
